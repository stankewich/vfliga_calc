        formation,

// –ó–∞–≥—Ä—É–∑–∫–∞ —Å–æ—Å—Ç–∞–≤–∞

    const key = `vs_lineup_${teamId}`;
    const data = localStorage.getItem(key);
    if (!data) return null;
    
    try {
        const lineup = JSON.parse(data);
        // –ü—Ä–æ–≤–µ—Ä–∫–∞ –∞–∫—Ç—É–∞–ª—å–Ω–æ—Å—Ç–∏ (–Ω–∞–ø—Ä–∏–º–µ—Ä, –Ω–µ —Å—Ç–∞—Ä—à–µ 7 –¥–Ω–µ–π)

            localStorage.removeItem(key);
            return null;
        }
        return lineup;
    } catch (e) {
        console.error('[Lineup] Error loading lineup:', e);
        return null;
    }
}

// –û—á–∏—Å—Ç–∫–∞ —Å–æ—Å—Ç–∞–≤–∞
